class Solution {
    public int jump(int[] nums) {
        if(nums==null ||nums.length<2) return 0;
        int cur=nums[0];
        int next=nums[0];
        int jumps=1;
        for(int i=0;i<nums.length;i++){
            next=Math.max(next,(nums[i]+i));
            if(i<nums.length-1 && i==cur){
                cur=next;
                jumps++;
                              }
            
            
        }
        return jumps;
    }
}
//tc:O(n)
//sc:O(1)
