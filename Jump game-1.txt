class Solution {
    public boolean canJump(int[] nums) {
        if(nums==null || nums.length==0) return true;
        int dest=nums.length-1;
        for(int i=dest-1;i>=0;i--){
           if(dest<=nums[i]+i) {
               dest=i;
           }
            System.out.println(dest);
        }
        return dest==0;
    }
}
//tc:O(n)
//sc:O(1)
